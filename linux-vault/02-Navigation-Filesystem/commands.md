# فصل ۲ - ناوبری و فایل‌سیستم | دستورات

## ناوبری پایه
```bash
pwd                     # نمایش پوشه جاری
cd /var/log             # رفتن به مسیر مطلق
cd ..                   # رفتن به پوشه والد
cd ~                    # رفتن به پوشه خانگی
cd -                    # برگشت به پوشه قبلی
```

## لیست فایل‌ها
```bash
ls                      # لیست ساده
ls -l                   # لیست با جزئیات
ls -la                  # شامل فایل‌های مخفی
ls -lh                  # حجم به صورت خوانا
ls -lt                  # مرتب بر اساس زمان
ls -lS                  # مرتب بر اساس حجم
ls -R                   # نمایش بازگشتی
ls *.conf               # فقط فایل‌های conf
```

## اطلاعات فایل
```bash
file /bin/bash          # نوع فایل
stat file.txt           # اطلاعات کامل (زمان، inode، مجوز)
tree /etc               # نمایش درختی پوشه
tree -L 2 /var          # فقط ۲ سطح عمق
```

## فایل‌های مخفی
```bash
ls -a                   # نمایش همه شامل مخفی
ls -la ~/.config        # محتوای پوشه مخفی
```
